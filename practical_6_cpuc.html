(a)-Write a program to check wheather the given number is odd or even. 

            
    C
#include
int main(){
    //to check even or odd
    int n;
    printf("Enter the number:");
    scanf("%d",&n);

    if(n%2 !=0){
        printf("This number is odd number");
    }else{printf("This is a even number");
    }

}

int main(){
