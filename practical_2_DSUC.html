Aim:-To write a program perform various operation on stack using Linked List in C language.

Requriment:- Personal computer or laptop ,Install Code editior with C Compiler and extensiton.

Theory:-
Stack:


  • Stack is a type of linear data structure.
  • It follow LIFO (Last In,First Out) principle.
  • Linked List implementation the size of stack is not fixed.
  • A pointer named top is used which points to the first node of the stack.
  • Push()----> Insert a new node at the top of the stack.
  • Pop()--->Deletr the top node from the stack.
  • Peek()--->View the top element without removing it.
  • Display--->Show all the elements of the stack.

Algorithm:-

    1.Push()
  • Create a new node.
  • If memory is full,display overflow condition.
  • Set newNode ->next = top
  • set top = newNode
    2.Pop()
  • Check wheather the stack is empty or not.
  • set temp = top
  • set top = top->next
  • Delete(free) the temporary node.
    3.Peek()
  • Start
  • Check if top==NULL
    • If true -->Display"Stack Underflow" or "Stack is Empty"
    • Go to Step 5
  • Else-->Display the value of top->data
  • End if
  • Stop

c 
    # include <stdio.h>
    # include <stdlib.h>

    struct Node{
        int data;
        struct Node* next;
    };
    struct Node* top =NULL;

    // push() operation 
    void push(int value){
        struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
        if(newNode == NULL){
        printf("Stack is Overflow condition\n");
        return ;
    }
    newNode->data = value;
    newNode->next = top;
    top = newNode ;
    printf("%d is pushed into stack",value);
}
// pop() operation 
void pop(){
    if(top == NULL){
        printf("Stack is underflow condition\n");
        return ;
    }
    struct Node* temp = top;
    printf("%d is Popped from stack\n",top->data);
    top = top->next;
    free(temp);
    
}
// peek() operation 
void peek(){
    if(top == NULL){
        printf("Stack is an empty\n");
    }else{
        printf("Top elements is %d\n",top->data);
    }
}

// dispaly
void dispaly(){
    struct Node* temp =top;
    if(temp == NULL){
        printf("Stack is Empty\n");
        return ;
    }
    printf("Stack elements are:\n");
    while(temp !=NULL){
        printf("%d\n",temp->data);
        temp = temp ->next;
    }
}

int main(){
    int choice,value;
while(1){ printf("\n--Stack Operation--\n");
    printf("\n1.Push ()");
    printf("\n2. Pop ()");
    printf("\n3. Peek ()");
    printf("\n4. Display");
    printf("\n5. Exit");
    printf("\nEnter the choice:");
    scanf("%d",&choice);

    switch (choice)
    {

    case 1: printf("Enter value:");
            scanf("%d",&value);
            push(value);
            break;
    case 2: pop();
            break;
    case 3: peek();
            break;
    case 4: dispaly();
            break;
    case 5: exit(0);

    default:
        printf("!Invalid Choice\n");
    }
}
}
