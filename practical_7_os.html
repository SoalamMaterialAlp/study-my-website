Aim:-Write a programs to simulate the following contiguous memory allocation techniques.
(a)- Worst Fit (b)- Best Fit (c)- First Fit
Requriment:- Personal Computer or Laptop,C compiler (Turbo c++ or Visual Studio code with extension).
Procedure:-
First Fit, Best Fit and Worst Fit contiguous memory allocation techniques 

             
    C  
    #include 

    void firstFit(int blockSize[], int m, int processSize[], int n) {
    int allocation[n];

    for (int i = 0; i < n; i++)
        allocation[i] = -1;

    printf("\n================ FIRST FIT STEP TABLE ================\n");
    printf("-------------------------------------------------------------\n");
    printf("| Process |Process Size | Block Allocated | Remaining Block Size   |\n");
    printf("-------------------------------------------------------------\n");

    for (int i = 0; i < n; i++) {
        int allocated = 0;
        for (int j = 0; j < m; j++) {
            if (blockSize[j] >= processSize[i]) {
                allocation[i] = j;
                printf("|   P%- 5d | %-4d       | Block %-2d     | %d - %d = %d        |\n",
                       i + 1, processSize[i], j + 1, blockSize[j], processSize[i], blockSize[j] - processSize[i]);
                blockSize[j] -= processSize[i];
                allocated = 1;
                break;
            }
        }
        if (!allocated) {
            printf("|   P%-5d | %-4d       | Not Allocated  | ---                 |\n", i + 1, processSize[i]);
        }
    }
    printf("-------------------------------------------------------------\n");

    printf("\nFinal First Fit Allocation:\n");
    printf("Process No.\tProcess Size\tBlock No.\n");
    for (int i = 0; i < n; i++) {
        printf("%d\t\t%d\t\t", i + 1, processSize[i]);
        if (allocation[i] != -1)
            printf("%d\n", allocation[i] + 1);
        else
            printf("Not Allocated\n");
    }
}


void bestFit(int blockSize[], int m, int processSize[], int n) {
    int allocation[n];

    for (int i = 0; i < n; i++)
        allocation[i] = -1;

    printf("\n================ BEST FIT STEP TABLE ================\n");
    printf("---------------------------------------------------------------------\n");
    printf("| Process |Process Size | Block Allocated (Best) | Remaining Block Size    |\n");
    printf("---------------------------------------------------------------------\n");

    for (int i = 0; i < n; i++) {
        int bestIdx = -1;
        for (int j = 0; j < m; j++) {
            if (blockSize[j] >= processSize[i]) {
                if (bestIdx == -1 || blockSize[j] < blockSize[bestIdx])
                    bestIdx = j;
            }
        }

        if (bestIdx != -1) {
            printf("|   P%-5d | %-4d       | Block %-3d            | %d - %d = %d         |\n",
                   i + 1, processSize[i], bestIdx + 1, blockSize[bestIdx], processSize[i], blockSize[bestIdx] - processSize[i]);
            allocation[i] = bestIdx;
            blockSize[bestIdx] -= processSize[i];
        } else {
            printf("|   P%-5d | %-4d       | Not Allocated        | ---                   |\n", i + 1, processSize[i]);
        }
    }
    printf("---------------------------------------------------------------------\n");

    printf("\nFinal Best Fit Allocation:\n");
    printf("Process No.\tProcess Size\tBlock No.\n");
    for (int i = 0; i < n; i++) {
        printf("%d\t\t%d\t\t", i + 1, processSize[i]);
        if (allocation[i] != -1)
            printf("%d\n", allocation[i] + 1);
        else
            printf("Not Allocated\n");
    }
}


void worstFit(int blockSize[], int m, int processSize[], int n) {
    int allocation[n];

    for (int i = 0; i < n; i++)

    printf("\n================ WORST FIT STEP TABLE ================\n");

    printf("---------------------------------------------------------------------\n");

    for (int i = 0; i < n; i++) {

                if (worstIdx == -1 || blockSize[j] > blockSize[worstIdx])
                    worstIdx = j;

        if (worstIdx != -1) {

            allocation[i] = worstIdx;
            blockSize[worstIdx] -= processSize[i];
        } else {
            printf("|   P%-5d | %-4d      | Not Allocated  | ---                  |\n", i + 1, processSize[i]);
        }

}
