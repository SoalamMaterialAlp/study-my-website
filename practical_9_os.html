(b). SCAN Disks Scheduling Algorithm


         
    C  
        #include 
#include 

int main() {
    int n, head, size, direction, total = 0;
    printf("SCAN disk sheduling Algorithm performed by Kadir Alam\n");
    printf("Enter number of disk requests: ");
    scanf("%d", &n);

    int req[n];
    printf("Enter the disk request sequence:\n ");
    for(int i = 0; i < n; i++)
        scanf("%d", &req[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter range/size disk(e.g., 200): ");
    scanf("%d", &size);

    printf("Enter direction (1 = RIGHT, 0 = LEFT): ");
    scanf("%d", &direction);

    // Sort requests
    for(int i = 0; i < n - 1; i++)
        for(int j = i + 1; j < n; j++)
            if(req[i] > req[j]) {
                int temp = req[i];
                req[i] = req[j];
                req[j] = temp;
            }

    // Find index where head fits
    int index = 0;
    for(int i = 0; i < n; i++)
        if(req[i] >= head) {
            index = i;
            break;
        }

    int order[200], k = 0; // To store the request order

    printf("\nSCAN Disk Movement:\n");

    if(direction == 1) { // RIGHT
        for(int i = index; i < n; i++) {
            printf("Head moves from %d to %d  Seek = %d\n", head, req[i], abs(head - req[i]));
            total += abs(head - req[i]);
            head = req[i];
            order[k++] = head;
        }

        printf("Head moves from %d to %d (Disk End) Seek = %d\n", head, size, abs(head - size));
        total += abs(head - size);
        head = size;

        for(int i = index - 1; i >= 0; i--) {
            printf("Head moves from %d to %d  Seek = %d\n", head, req[i], abs(head - req[i]));
            total += abs(head - req[i]);
            head = req[i];
            order[k++] = head;
        }
    }
    else { // LEFT
        for(int i = index - 1; i >= 0; i--) {
            printf("Head moves from %d to %d  Seek = %d\n", head, req[i], abs(head - req[i]));
            total += abs(head - req[i]);
            head = req[i];
            order[k++] = head;
        }

        printf("Head moves from %d to 0 (Disk Start) Seek = %d\n", head, head);
        total += head;
        head = 0;

        for(int i = index; i < n; i++) {
            printf("Head moves from %d to %d  Seek = %d\n", head, req[i], abs(head - req[i]));
            total += abs(head - req[i]);
            head = req[i];
            order[k++] = head;
        }
    }

    printf("\n=== SCAN Request Order ===\n");
    for(int i = 0; i < k; i++) {
        if(i == k-1) printf("%d\n", order[i]);
        else printf("%d ", order[i]);
    }


    printf("\n\nTotal Seek Time = %d\n", total);
    printf("\n\nAverage Seek Time = %.2f\n",(float) total/n);

    return 0;
}

(c). C-SCAN Disks Scheduling Algorithm

int main() {
    int n, head, size, total = 0;
     printf("C-SCAN Disak Sheduling Algorithm performed by Kadir Alam\n");
    printf("Enter number of disk requests: ");
    scanf("%d", &n);

    int req[n];
    printf("Enter the disk request sequence: \n");
    for(int i = 0; i < n; i++)
        scanf("%d", &req[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter disk end cylinder number (e.g., 200): ");
    scanf("%d", &size);   // Now size = last cylinder (not size-1)

    // Sort the requests
    for(int i = 0; i < n - 1; i++)
        for(int j = i + 1; j < n; j++)
            if(req[i] > req[j]) {
                int temp = req[i];
                req[i] = req[j];
                req[j] = temp;
            }

    // Find where head fits
    int index = 0;
    for(int i = 0; i < n; i++) {
        if(req[i] >= head) {
            index = i;
            break;
        }
    }

    int order[200], k = 0;

    printf("\nC-SCAN Head Movement:\n");

    // Move RIGHT first
    for(int i = index; i < n; i++) {
        printf("Head moves from %d to %d  Seek = %d\n", head, req[i], abs(head - req[i]));
        total += abs(head - req[i]);
        head = req[i];
        order[k++] = head;
    }

    // Move to disk END
    printf("Head moves from %d to %d (Disk End)  Seek = %d\n", head, size, abs(head - size));
    total += abs(head - size);
    head = size;

    // Jump to 0 (NO Seek counted because it is circular)
    printf("Head jumps from %d to 0 (No Seek Count)\n", head);
    head = 0;

    // Now move again from start
    for(int i = 0; i < index; i++) {
        printf("Head moves from %d to %d  Seek = %d\n", head, req[i], abs(head - req[i]));
        total += abs(head - req[i]);
        head = req[i];
        order[k++] = head;
    }

    // Print Request Order
    printf("\n=== C-SCAN Request Order ===\n");
    for(int i = 0; i < k; i++) {
        if(i == k-1) printf("%d\t", order[i]);
        else printf("%d ", order[i]);
    }

    // Total and Average Seek Time
    printf("\n\nTotal Seek Time = %d\n", total);
    printf("Average Seek Time = %.2f\n", (float)total / k);

    return 0;
}
