Aim:-To write a program perform various operation on stack using Linked List in C language.

Requriment:- Personal computer or laptop ,Install Code editior with C Compiler and extensiton.

Theory:-

Stack:

  • Stack is a type of linear data structure.
  • It follow LIFO (Last In,First Out) principle.
    Stack Operation:
  1. Push ()
  2. Pop()
  3. Peek ()
  4. IsFull()
  5. IsEmpty()

Algorithm:-

    1.Push()
  • If top == MAX-1 --->Stack Full.
  • Else top = +1
  • stack [top] = value

    2.Pop()
  • If top ==-1 --> Stack Empty.
  • Else remove stack[top]
  • top = top-1

    3.Peek()
  • If top ==-1 -->stack empty
  • Else displaystack[top]

    4.IsFull()
  • If top == MAX-1 -->Full
  • Else Not Full

    5.IsEmpty()
  • If top==-1 -->Empty

c 

#include <stdio.h>
int stack[100];
int top = -1;
int MAX;

//PUSH OPERATION
void push(int x){
 if (top == MAX - 1){
        printf("Stack is overflow condition!Cannot Push \n");
    }
else{
        top++;
        stack[top] = x;
        printf("%d pushed into stack\n",x);
    }
}
//POP OPERATION
void pop(){
    if(top==-1){
        printf("Stack is underflow condition!Stack is empty.\n");
    }else{
        printf("%d popped from stack\n",stack[top]);
        top--;
    }
}
//PEEK OPERATION
void peek(){
    if(top==-1){
        printf("Stack is empty.\n");
    }else{
        printf("Top Element is %d\n",stack[top]);
    }
}
//isFull OPERATION
void isFull(){
    if(top == MAX - 1){
        printf("Stack is full.");
    }else{
        printf("Stack is not full");
    }
}
//ISEMPTY OPERATION
void IsEmpty(){
    if(top == - 1){
        printf("Stack is empty.");
    }else{
        printf("Stack is not empty.");
    }
}
//DISPLAY OPERATION
void display(){
    if(top==-1){
        printf("Stack is Empty\n");
    }else{
        printf("Stack elements are \n");
        for(int i=top;i>=0;i--){
           printf("%d\n",stack[i]);
        }
    }
}
